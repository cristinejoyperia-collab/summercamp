        function playChime() {
            try {
                const AudioContext = window.AudioContext || window.webkitAudioContext;
                const ctx = new AudioContext();
                const osc1 = ctx.createOscillator();
                const gain1 = ctx.createGain();
                osc1.type = 'sine';
                osc1.frequency.setValueAtTime(587.33, ctx.currentTime); 
                gain1.gain.setValueAtTime(0.2, ctx.currentTime);
                gain1.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.3);
                osc1.connect(gain1); gain1.connect(ctx.destination);
                osc1.start(); osc1.stop(ctx.currentTime + 0.3);

                setTimeout(() => {
                    const osc2 = ctx.createOscillator();
                    const gain2 = ctx.createGain();
                    osc2.type = 'sine';
                    osc2.frequency.setValueAtTime(880, ctx.currentTime); 
                    gain2.gain.setValueAtTime(0.2, ctx.currentTime);
                    gain2.gain.exponentialRampToValueAtTime(0.01, ctx.currentTime + 0.4);
                    osc2.connect(gain2); gain2.connect(ctx.destination);
                    osc2.start(); osc2.stop(ctx.currentTime + 0.4);
                }, 100);
            } catch (e) { console.log("Audio contexts blocked."); }
        }

        function initCabinMap() {
            const mapElement = document.getElementById('airplane-cabin-map');
            const seatLetters = ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'J', 'K'];
            
            for (let i = 1; i <= 72; i++) {
                const rowDiv = document.createElement('div');
                rowDiv.className = 'cabin-row';
                
                const label = document.createElement('div');
                label.className = 'row-num'; label.innerText = i;
                rowDiv.appendChild(label);
                
                seatLetters.forEach((letter, index) => {
                    if (index === 3 || index === 7) {
                        const aisle = document.createElement('div');
                        aisle.className = 'cabin-aisle';
                        rowDiv.appendChild(aisle);
                    }
                    
                    const seat = document.createElement('div');
                    seat.className = 'plane-seat';
                    seat.innerText = `${i}${letter}`;
                    seat.setAttribute('data-seat-id', `${i}${letter}`);
                    
                    seat.addEventListener('click', function() {
                        playChime();
                        document.querySelectorAll('.plane-seat').forEach(s => s.classList.remove('selected'));
                        this.classList.add('selected');
                        selections.seat = this.getAttribute('data-seat-id');
                        document.getElementById('seat-selection-status').innerText = `🎉 Target Seat Claimed: ${selections.seat}`;
                    });
                    rowDiv.appendChild(seat);
                });
                mapElement.appendChild(rowDiv);
            }
        }

        function nextStep(currentStepNum) {
            if (currentStepNum === 1) {
                const name = document.getElementById('student-name').value.trim();
                if (!name) { alert('Please enter your Full Name first! 🕒'); return; }
                if (!selections.gender) { alert('Please select your Gender! 👦👧'); return; }
            } else if (currentStepNum === 2 && !selections.airline) {
                alert('Please select an Airline Carrier to continue! ✈️'); return;
            } else if (currentStepNum === 3 && !selections.country) {
                alert('Please pick your Destination Country to continue! 🗺️'); return;
            } else if (currentStepNum === 4 && !selections.seat) {
                alert('Please select your preferred Cabin Seat from the map! 💺'); return;
            } else if (currentStepNum === 5 && !selections.weight) {
                alert('Please choose a Max Luggage Weight option! 🧳'); return;
            } else if (currentStepNum === 6) {
                if(!selections.food) { alert('Please choose your food option meal! 🍝'); return; }
                if(!selections.drink) { alert('Please select your target beverage drink! 🥤'); return; }
            }

            playChime();
            if (currentStepNum === 1) { stopCamera(); }
            document.getElementById(`step-${currentStepNum}`).classList.remove('active');
            document.getElementById(`step-${currentStepNum + 1}`).classList.add('active');
        }
